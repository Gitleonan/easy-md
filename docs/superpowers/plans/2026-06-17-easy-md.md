# md++ (easy-md) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建一个 Windows 平台轻量便捷的 Markdown 纯阅读器（应用名 md++ / 项目名 easy-md），双栏布局、极简风格、支持全语法（含 Mermaid/KaTeX/shiki）、多标签、深浅主题、图片 lightbox、全文搜索、四种导出、文件监听自动刷新，架构预留 macOS 适配。

```ts
import { getHighlighter, type Highlighter } from 'shiki';

const themes = { light: 'github-light', dark: 'github-dark' } as const;
export type ThemeName = keyof typeof themes;

let highlighterPromise: Promise<Highlighter> | null = null;

async function getHL(): Promise<Highlighter> {
  if (!highlighterPromise) {
    highlighterPromise = getHighlighter({ themes: ['github-light', 'github-dark'], langs: [] });
  }
  return highlighterPromise;
}

const CODE_BLOCK_RE = /<pre><code class="language-([\w-]+)">(.*?)<\/code><\/pre>/gs;

export async function highlightCodeBlocks(html: string, theme: ThemeName): Promise<string> {
  const hl = await getHL();
  return html.replace(CODE_BLOCK_RE, (_, lang, encoded) => {
    const code = decodeHtmlEntities(encoded);
    try {
      if (!hl.getLanguage(lang)) {
        // 不支持的语言：动态加载一次
        return `<pre class="shiki"><code>${code}</code></pre>`;
      }
      return hl.codeToHtml(code, { lang, theme: themes[theme] });
    } catch {
      return `<pre class="shiki"><code>${code}</code></pre>`;
    }
  });
}

function decodeHtmlEntities(s: string): string {
  return s
    .replace(/&lt;/g, '<')
    .replace(/&gt;/g, '>')
    .replace(/&quot;/g, '"')
    .replace(/&#39;/g, "'")
    .replace(/&amp;/g, '&');
}
```

```ts
export interface MermaidBlock {
  id: string;
  source: string;
}

const MERMAID_RE = /<div class="mermaid">([\s\S]*?)<\/div>/g;

/** 从渲染后的 HTML 提取 mermaid 块源码，DOM 渲染阶段用 mermaid.render 异步替换为 SVG */
export function extractMermaidBlocks(html: string): MermaidBlock[] {
  const blocks: MermaidBlock[] = [];
  let m: RegExpExecArray | null;
  let i = 0;
  while ((m = MERMAID_RE.exec(html)) !== null) {
    blocks.push({ id: `mermaid-${i++}`, source: m[1].trim() });
  }
  return blocks;
}

/** 异步把指定容器内的 mermaid div 替换为 SVG（在 DOM 注入后调用） */
export async function renderMermaidInContainer(container: HTMLElement): Promise<void> {
  const { default: mermaid } = await import('mermaid');
  mermaid.initialize({ startOnLoad: false, theme: document.documentElement.getAttribute('data-theme') === 'dark' ? 'dark' : 'default' });
  const nodes = container.querySelectorAll<HTMLElement>('.mermaid');
  for (const node of Array.from(nodes)) {
    const source = node.textContent || '';
    try {
      const id = `m-${Math.random().toString(36).slice(2)}`;
      const { svg } = await mermaid.render(id, source);
      node.innerHTML = svg;
    } catch {
      node.innerHTML = '<div style="color:#cc0000">Mermaid diagram error</div>';
    }
  }
}
```

```ts
import type MarkdownIt from 'markdown-it';
import { md } from './plugins';
import type { TocItem } from '../../types';

/** 在 markdown-it 解析时收集 heading token，返回扁平 TOC，前端再构树 */
export function extractToc(source: string): TocItem[] {
  const tokens = md.parse(source, {});
  const items: TocItem[] = [];
  const seen = new Map<string, number>();

  for (let i = 0; i < tokens.length; i++) {
    const t = tokens[i];
    if (t.type === 'heading_open') {
      const level = Number(t.tag.slice(1)) as TocItem['level'];
      // 下一个 inline token 是标题文本
      const inline = tokens[i + 1];
      const text = inline?.content ?? '';
      const id = slugify(text, seen);
      items.push({ id, level, text });
    }
  }
  return items;
}

function slugify(text: string, seen: Map<string, number>): string {
  const base = text
    .toLowerCase()
    .replace(/[^\w\s-]/g, '')
    .replace(/\s+/g, '-')
    .replace(/-+/g, '-')
    .trim();
  const count = seen.get(base) ?? 0;
  seen.set(base, count + 1);
  return count === 0 ? base : `${base}-${count}`;
}
```

```rust
use base64::{engine::general_purpose, Engine as _};

#[tauri::command]
pub async fn read_text_file(path: String) -> Result<String, String> {
    fs::read_to_string(&path).map_err(|e| format!("读取文件失败: {}", e))
}

/// 解析图片：网络地址原样返回，本地路径读为 data URL；失败返回占位符
#[tauri::command]
pub async fn resolve_image(md_file_path: String, src: String) -> Result<String, String> {
    if src.starts_with("http://") || src.starts_with("https://") {
        return Ok(src);
    }
    let img_path = resolve_local_path(&md_file_path, &src);
    match fs::read(&img_path) {
        Ok(bytes) => {
            let mime = guess_mime(&img_path);
            let b64 = general_purpose::STANDARD.encode(&bytes);
            Ok(format!("data:{};base64,{}", mime, b64))
        }
        Err(_) => Ok(PLACEHOLDER_DATA_URL.to_string()),
    }
}

pub fn resolve_local_path(md_file_path: &str, src: &str) -> PathBuf {
    let p = Path::new(src);
    if p.is_absolute() {
        PathBuf::from(src)
    } else {
        Path::new(md_file_path)
            .parent()
            .unwrap_or(Path::new("."))
            .join(src)
    }
}

pub fn guess_mime(path: &Path) -> &'static str {
    match path.extension().and_then(|e| e.to_str()) {
        Some("png") => "image/png",
        Some("jpg") | Some("jpeg") => "image/jpeg",
        Some("gif") => "image/gif",
        Some("webp") => "image/webp",
        Some("svg") => "image/svg+xml",
        _ => "application/octet-stream",
```

```ts
export async function resolveImages(container: HTMLElement, mdFilePath: string): Promise<void> {
  const imgs = Array.from(container.querySelectorAll<HTMLImageElement>('img'));
```

```ts
import { create } from 'zustand';
import type { Tab, TocItem } from '../types';
import { readFile } from '../ipc/files';
import { renderMarkdown } from '../features/markdown/render';
import { highlightCodeBlocks } from '../features/markdown/highlight';
import { extractToc } from '../features/markdown/toc';

interface TabsState {
  tabs: Tab[];
  activeTabId: string | null;
  openTab: (filePath: string) => Promise<void>;
  closeTab: (id: string) => void;
  setActive: (id: string) => void;
  setScrollTop: (id: string, scrollTop: number) => void;
  updateSource: (id: string, source: string, html: string, toc: TocItem[]) => void;
}

let seq = 0;

async function buildTab(filePath: string): Promise<Tab> {
  const source = await readFile(filePath);
  const toc = extractToc(source);
  const rawHtml = renderMarkdown(source);
  const html = await highlightCodeBlocks(rawHtml, 'light'); // 主题后续从 themeStore 读
  const fileName = filePath.split(/[/\\]/).pop() || filePath;
  return {
    id: `tab-${++seq}`,
    filePath,
    fileName,
    source,
    html,
    toc,
    scrollTop: 0,
    tocExpanded: {},
  };
}

export const useTabsStore = create<TabsState>((set, get) => ({
  tabs: [],
  activeTabId: null,

  async openTab(filePath) {
    const normalized = filePath.replace(/\//g, '\\');
    const existing = get().tabs.find((t) => t.filePath === normalized || t.filePath === filePath);
    if (existing) {
      set({ activeTabId: existing.id });
      return;
    }
    const tab = await buildTab(filePath);
    set((s) => ({ tabs: [...s.tabs, tab], activeTabId: tab.id }));
  },

  closeTab(id) {
    set((s) => {
      const idx = s.tabs.findIndex((t) => t.id === id);
      const tabs = s.tabs.filter((t) => t.id !== id);
      let activeTabId = s.activeTabId;
      if (activeTabId === id) {
        activeTabId = tabs[idx]?.id ?? tabs[idx - 1]?.id ?? null;
      }
      return { tabs, activeTabId };
    });
  },

  setActive(id) {
    set({ activeTabId: id });
  },

  setScrollTop(id, scrollTop) {
    set((s) => ({ tabs: s.tabs.map((t) => (t.id === id ? { ...t, scrollTop } : t)) }));
  },

  updateSource(id, source, html, toc) {
    set((s) => ({ tabs: s.tabs.map((t) => (t.id === id ? { ...t, source, html, toc } : t)) }));
  },
}));
```

```ts
import { useEffect } from 'react';
import { getCurrentWebview } from '@tauri-apps/api/webview';
import { useTabsStore } from '../stores/tabsStore';
import { pickMdFiles } from '../ipc/system';

const MD_EXT = /\.(md|markdown)$/i;

export function normalizeDroppedPaths(files: { path: string; name: string }[]): string[] {
  return files.filter((f) => MD_EXT.test(f.name)).map((f) => f.path);
}

export function useOpenFile() {
  const openTab = useTabsStore((s) => s.openTab);

  // 拖拽
  useEffect(() => {
    const unlisten = getCurrentWebview().onDragDropEvent(async (e) => {
      if (e.payload.type === 'drop') {
        const paths = (e.payload.paths || []).filter((p) => MD_EXT.test(p));
        for (const p of paths) await openTab(p);
      }
    });
    return () => { unlisten.then((u) => u()); };
  }, [openTab]);
}

export async function openViaDialog() {
  const paths = await pickMdFiles();
  if (!paths) return;
  for (const p of paths) {
    await useTabsStore.getState().openTab(p);
  }
}
```

```tsx
import type { TocItem } from '../../types';
import { useTabsStore } from '../../stores/tabsStore';

export interface TocNode extends TocItem {
  children?: TocNode[];
}

export function buildTocTree(flat: TocItem[]): TocNode[] {
  const root: TocNode[] = [];
  const stack: TocNode[] = [];
  for (const item of flat) {
    const node: TocNode = { ...item };
    while (stack.length && stack[stack.length - 1].level >= node.level) stack.pop();
    if (stack.length === 0) {
      root.push(node);
    } else {
      const parent = stack[stack.length - 1];
      parent.children = parent.children || [];
      parent.children.push(node);
    }
    stack.push(node);
  }
  return root;
}

export function Sidebar() {
  const tab = useTabsStore((s) => s.tabs.find((t) => t.id === s.activeTabId));
  if (!tab) return null;
  const tree = buildTocTree(tab.toc);

  const renderNodes = (nodes: TocNode[]) => (
    <ul>
      {nodes.map((n) => (
        <li key={n.id}>
          <a href={`#${n.id}`} style={{ paddingLeft: (n.level - 1) * 12 }}>
            {n.text}
          </a>
          {n.children && renderNodes(n.children)}
        </li>
      ))}
    </ul>
  );

  return <aside className="sidebar">{renderNodes(tree)}</aside>;
}
```

```ts
type Resolved = 'light' | 'dark';

interface ThemeState {
  theme: ThemeMode;
  resolved: Resolved;
  setTheme: (t: ThemeMode) => void;
  toggle: () => void;
}

function getSystemTheme(): Resolved {
  return window.matchMedia?.('(prefers-color-scheme: dark)').matches ? 'dark' : 'light';
}

function resolve(mode: ThemeMode): Resolved {
  return mode === 'system' ? getSystemTheme() : mode;
}

    set({ theme, resolved: r });

  toggle() {

    get().setTheme(next);

}));
```

```ts
/** 在渲染后的 DOM 上为 h1-h6 注入 id，与 TOC 提取的 slug 算法保持一致 */
export function injectAnchors(container: HTMLElement, toc: TocItem[]): void {
  const headings = Array.from(container.querySelectorAll<HTMLElement>('h1,h2,h3,h4,h5,h6'));
  const seen = new Map<string, number>();
  for (const h of headings) {

    const base = text.toLowerCase().replace(/[^\w\s-]/g, '').replace(/\s+/g, '-');
    const count = seen.get(base) ?? 0;

}
```

```ts
const MARK_CLASS = 'search-mark';
const MARK_CURRENT_CLASS = 'search-mark-current';

export interface SearchSession {
  matches: HTMLElement[];
  current: number;
}

/** 在 root 内所有文本节点中查找 keyword，包 <mark>，返回匹配节点列表 */
export function highlightMatches(root: HTMLElement, keyword: string): number {
  clearHighlights(root);
  if (!keyword) return 0;
  const kwLower = keyword.toLowerCase();
  const walker = document.createTreeWalker(root, NodeFilter.SHOW_TEXT, {
    acceptNode(node) {
      const p = node.parentElement;
      if (!p) return NodeFilter.FILTER_REJECT;
      if (p.tagName === 'SCRIPT' || p.tagName === 'STYLE') return NodeFilter.FILTER_REJECT;
      return node.nodeValue && node.nodeValue.toLowerCase().includes(kwLower)
        ? NodeFilter.FILTER_ACCEPT
        : NodeFilter.FILTER_REJECT;
    },
  });

  const textNodes: Text[] = [];
  let n: Node | null;
  while ((n = walker.nextNode())) textNodes.push(n as Text);

  let total = 0;
  for (const tn of textNodes) {
    const text = tn.nodeValue || '';
    const lower = text.toLowerCase();
    let i = 0;
    const frag = document.createDocumentFragment();
    let idx = lower.indexOf(kwLower, i);
    while (idx !== -1) {
      if (idx > i) frag.appendChild(document.createTextNode(text.slice(i, idx)));
      const mark = document.createElement('mark');
      mark.className = MARK_CLASS;
      mark.textContent = text.slice(idx, idx + keyword.length);
      frag.appendChild(mark);
      total++;
      i = idx + keyword.length;
      idx = lower.indexOf(kwLower, i);
    }
    if (i < text.length) frag.appendChild(document.createTextNode(text.slice(i)));
    tn.parentNode?.replaceChild(frag, tn);
  }
  return total;
}

export function clearHighlights(root: HTMLElement): void {
  const marks = root.querySelectorAll(`mark.${MARK_CLASS}, mark.${MARK_CURRENT_CLASS}`);
  marks.forEach((m) => {
    const text = document.createTextNode(m.textContent || '');
    m.parentNode?.replaceChild(text, m);
  });
  root.normalize();
}

export function focusMatch(marks: HTMLElement[], index: number): void {
  marks.forEach((m, i) => m.classList.toggle(MARK_CURRENT_CLASS, i === index));
  if (marks[index]) marks[index].scrollIntoView({ block: 'center', behavior: 'smooth' });
}
```

```rust
use notify_debouncer_mini::{new_debouncer, DebouncedEvent};
use std::{path::PathBuf, sync::Mutex, time::Duration};
use tauri::{AppHandle, Emitter, Manager};

pub struct WatcherState {
    pub watcher: Mutex<Option<notify_debouncer_mini::Debouncer<notify::FsEventWatcher>>>,
}

pub fn watch_paths(app: AppHandle, paths: Vec<String>) {
    let app_handle = app.clone();
    let state = app.state::<WatcherState>();
    let mut debouncer = new_debouncer(Duration::from_millis(300), move |res: Result<Vec<DebouncedEvent>, _>| {
        if let Ok(events) = res {
            let changed: Vec<String> = events
                .iter()
                .filter_map(|e| e.path.to_str().map(|s| s.to_string()))
                .collect();
            if !changed.is_empty() {
                let _ = app_handle.emit("file-changed", &changed);
            }
        }
    }).expect("create debouncer");

    for p in &paths {
        if let Err(e) = debouncer.watcher().watch(PathBuf::from(p).as_path(), RecursiveMode::NonRecursive) {
            eprintln!("watch failed {:?}: {}", p, e);
        }
    }

    *state.watcher.lock().unwrap() = Some(debouncer);
}
```

```ts
          if (source === tab.source) continue; // 内容没变跳过
          const toc = extractToc(source);
          const html = await highlightCodeBlocks(renderMarkdown(source), theme);
          updateSource(tab.id, source, html, toc);
        } catch (err) {
          console.error('reload failed', err);
        }
      }
    });
    return () => { unlisten.then((u) => u()); };
  }, []);
}
```

```ts
import { writeTextFile } from '@tauri-apps/plugin-fs';

export function buildStandaloneHtml(contentHtml: string, css: string, theme: 'light' | 'dark'): string {
  return `<!DOCTYPE html>
<html lang="zh" data-theme="${theme}">
<head>
<meta charset="UTF-8">
<title>md++ export</title>
<style>${css}</style>
</head>
<body class="md-content">${contentHtml}</body>
</html>`;

export async function exportHtml(contentHtml: string, theme: 'light' | 'dark'): Promise<void> {
  const css = await fetchThemeCss();
  const full = buildStandaloneHtml(contentHtml, css, theme);
  const path = await save({ defaultPath: 'export.html', filters: [{ name: 'HTML', extensions: ['html'] }] });
  if (!path) return;
  await writeTextFile(path, full);
}

async function fetchThemeCss(): Promise<string> {
  // 读取打包后的 themes.css 与 prose.css 文本（构建时把两份 css 内联进来）
  // 简化：直接 import 字符串
  return THEMES_CSS + '\n' + PROSE_CSS;

const THEMES_CSS = `:root{--bg:#fff;--fg:#1f2328;--bg-secondary:#f6f8fa;--fg-muted:#656d76;--border:#d0d7de;--accent:#0969da;--code-bg:#f6f8fa;--mark-bg:#fff8c5}

.md-content pre{background:var(--code-bg);padding:16px;border-radius:6px;overflow:auto}
.md-content img{max-width:100%}`;

/** 调用浏览器原生打印（含"打印到 PDF"） */
export async function printContent(el: HTMLElement): Promise<void> {
  const printWin = window.open('', '_blank');
  if (!printWin) return;
  const css = await fetchThemeCss();
  printWin.document.write(buildStandaloneHtml(el.innerHTML, css, document.documentElement.getAttribute('data-theme') as 'light' | 'dark'));
  printWin.document.close();
  printWin.focus();
  printWin.print();
}

/** 复制渲染后的富文本到剪贴板 */
export async function copyRichContent(el: HTMLElement): Promise<void> {
  const html = el.innerHTML;
  const text = el.textContent || '';
  const clipboardItem = new ClipboardItem({
    'text/html': new Blob([html], { type: 'text/html' }),
    'text/plain': new Blob([text], { type: 'text/plain' }),
  });
  await navigator.clipboard.write([clipboardItem]);
}
```

```rust
use serde::{Deserialize, Serialize};
use std::fs;
use tauri::Manager;

#[derive(Serialize, Deserialize, Clone)]
pub struct RecentFile {
    pub path: String,
    pub name: String,
    pub last_opened_at: i64,
}

fn recent_path(app: &tauri::AppHandle) -> std::path::PathBuf {
    let dir = app.path().app_data_dir().expect("no app_data_dir");
    let _ = fs::create_dir_all(&dir);
    dir.join("recent.json")
}

#[tauri::command]
pub async fn list_recent(app: tauri::AppHandle) -> Result<Vec<RecentFile>, String> {
    let p = recent_path(&app);
    if !p.exists() { return Ok(vec![]); }
    let s = fs::read_to_string(p).map_err(|e| e.to_string())?;
    serde_json::from_str(&s).map_err(|e| e.to_string())
}

#[tauri::command]
pub async fn add_recent(app: tauri::AppHandle, file: RecentFile) -> Result<(), String> {
    let p = recent_path(&app);
    let mut list: Vec<RecentFile> = if p.exists() {
        serde_json::from_str(&fs::read_to_string(&p).unwrap_or_default()).unwrap_or_default()
    } else { vec![] };
    list.retain(|f| f.path != file.path);
    list.insert(0, file);
    list.truncate(20);
    fs::write(&p, serde_json::to_string_pretty(&list).unwrap()).map_err(|e| e.to_string())?;
    Ok(())
}
```
